引入layui

cdn引入layui,创建index.html和index.js。 

        <head>
        <!--index.html的内容（index.html和index.js在同一文件夹情况下。）-->
               <link rel="stylesheet" href="//unpkg.com/layui@2.6.8/dist/css/layui.css">
               </head>
               <body>
               <script src="//unpkg.com/layui@2.6.8/dist/layui.js"></script>
               <layui.config({
            base: '/' //你的index.js所在位置
          }).use('index'); //加载index.js>
          </body>
    
        /**
  index.js 项目 JS 主入口
  以依赖 layui 的 layer 和 form 模块为例
**/

layui.define(["layer", "form", "code", "element"], function (exports) {
  var layer = layui.layer,
    form = layui.form,
    code = layui.code,
    element = layui.element;

  layer.msg("Hello，欢迎");

  exports("index", {}); //注意，这里是模块输出的核心，模块名必须和 use 时的模块名一致
});
//这是index.js的内容，如果在运行inde.html出现“Hello，欢迎”证明引入成功。

本地引入

下载layui包

index.js内容不用改动，index.html内容如下。 

        <head>
        <!--index.html的内容（index.html和index.js在同一文件夹情况下，layui文件在主目录里的layui文件夹。）-->
               <link rel="stylesheet" href="./layui/css/layui.css""><!--href后面是你的layui文件的位置。-->
               </head>
               <body>
               <script src="./layui/layui.js"></script><!--src后面是你的layui文件位置。-->
               <layui.config({
            base: '/' //你的index.js所在位置
          }).use('index'); //加载index.js>
          </body>
